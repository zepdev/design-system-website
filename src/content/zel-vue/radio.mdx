# Radio

A radio button is a graphical control element that allows users to only select one option from a group of options at a time. Radio buttons must always contain a clear and explicit label which appear on the right side of the button.

## Example

```js
<template>
  <div id="my-component">
    <zRadioButton
      name="title"
      id="zep_radio_male"
      class="zep-radio__input"
      label="Male"
      value="male"
      v-bind:checked="true"
      v-bind:required="true"
      v-on:input="changeValue"
    ></zRadioButton>
    <zRadioButton
      name="title"
      id="zep_radio_female"
      class="zep-radio__input"
      label="Female"
      value="female"
      v-bind:checked="false"
      v-bind:required="true"
      v-on:input="changeValue"
    ></zRadioButton>
    <span>Selected Radio: {{radioButtonCheck}}</span>
  </div>
</template>

<script>

import { ZelRadioButton } from "@zlab-de/zel-vue";

export default {
  name: "my-component",
  components: {
   ZelVueRadioButton,
  },
  data: function() {
    return {
      radioButtonCheck: "",
    };
  },
  methods: {
    changeValue: function(newValue) {
      this.radioButtonCheck = newValue;
    }
  }
};
</script>
```
