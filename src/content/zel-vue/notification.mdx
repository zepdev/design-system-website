## Example

```js
<template>
    <div id="my-component">
        <zel-vue-notification variant="danger">
            Access Denied
        </zel-vue-notification>
    </div>
</template>

<script>
import { ZelVueNotifications } from "@zlab-de/zel-vue";

export default {
  name: "my-component",
  components: {
    ZelVueNotifications,
  }
};
</script>
```
