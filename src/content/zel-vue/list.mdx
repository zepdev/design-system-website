# List

Lists group related content together and should be used to present simple pieces of information. More complex data sets should be displayed with a table element.

---

## Example

```js
<template>
  <div id="my-component">
    <zel-vue-list :dataSource="this.bulletItems" variant="bullet" />
    <zel-vue-list :dataSource="this.attributeItems" variant="attribute" />
  </div>
</template>

<script>

import { ZelVueList } from "@zlab-de/zel-vue";

export default {
  name: "my-component",
  components: {
    ZelVueList
  },
  data: function() {
    return {
      bulletItems: [{ text: "item 1" }, { text: "item 2" }, { text: "item 3" }],
      attributeItems: [
        { label: "label 1", text: "item 1" },
        { label: "label 2", text: "item 2" },
        { label: "label 3", text: "item 3" }
      ]
    };
  }
};
</script>
```
