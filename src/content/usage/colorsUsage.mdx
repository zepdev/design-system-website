# Developers

If you are using theme.json colors can be acessed from theme.color.[section].[name].[hex or rgb]

For example,

```jsx
import React from 'react'
import theme from '@zlab-de/zel-react/theme.json'

function MyComponent() {
  return <p style={{ color: theme.color.zlab.cyan.hex }}>I am zLab Cyan!</p>
}

export default MyComponent
```

An example using React-JSS in Create-React-App;

```jsx
// index.js
import React from 'react'
import ReactDOM from 'react-dom'
import App from './App'
import { ThemeProvider } from 'react-jss'
import '@zlab-de/zel-react/zel.css'
import guidelines from '@zlab-de/zel-react/guidelines.json'
import * as serviceWorker from './serviceWorker'
import './index.css'

ReactDOM.render(
  <ThemeProvider theme={guidelines}>
    <App />
  </ThemeProvider>,
  document.getElementById('root')
)

// If you want your app to work offline and load faster, you can change
// unregister() to register() below. Note this comes with some pitfalls.
// Learn more about service workers: https://bit.ly/CRA-PWA
serviceWorker.unregister()

// your component
import React from 'react'
import { createUseStyles } from 'react-jss'

const useStyles = createUseStyles(theme => ({
  text: {
    textAlign: 'center',
    color: theme.color.cat.catYellow.hex,
    background: theme.color.gray.grayLighter.hex,
    padding: `${theme.spacing.component.s.rem}rem`,
  },
}))

function MyComponent({ ...props }) {
  const classes = useStyles(props)
  return (
    <p className={classes.text}>I am Cat Yellow text on a gray background!</p>
  )
}

export default MyComponent
```
