```
import React, { useState } from 'react'
import Radio from '@zlab-de/zel-react/Radio'
import RadioGroup from '@zlab-de/zel-react/RadioGroup'

function RadioExample() {
  const [radio, setRadio] = useState('a')
  const [radioInline, setRadioInline] = useState('c')


  return (
    <div>
      <RadioGroup title="Select One">
        <Radio
          value="a"
          id="r1"
          name="value 1"
          checked={radio === 'a'}
          onChange={e=>setRadio(e.target.value)}
        />
        <Radio
          value="b"
          id="r2"
          name="value 2"
          checked={radio === 'b'}
          onChange={e=>setRadio(e.target.value)}
        />
      </RadioGroup>
          <RadioGroup title="Select One">
        <Radio
          variant="inline"
          value="c"
          id="r3"
          name="value 3"
          checked={radioInline === 'c'}
          onChange={e=>setRadioInline(e.target.value)}
        />
        <Radio
          variant="inline"
          value="d"
          id="r4"
          name="value 4"
          checked={radioInline === 'd'}
          onChange={e=>setRadioInline(e.target.value)}
        />
      </RadioGroup>
    </div>
  )
}

export default RadioExample
```
