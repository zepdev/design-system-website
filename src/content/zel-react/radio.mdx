# Radio

A radio button is a graphical control element that allows users to only select one option from a group of options at a time. Radio buttons must always contain a clear and explicit label which appear on the right side of the button.

## Example

<div style={{ marginBottom: 30 }}>
  <RadioExample />
</div>

```
import React, { useState } from 'react'
import Radio from '@zlab-de/zel-react/Radio'
import RadioGroup from '@zlab-de/zel-react/RadioGroup'

function RadioExample() {
  const [radio, setRadio] = useState('a')
  function handleChange(event) {
    setRadio(event.target.value)
  }

  return (
    <div>
      <RadioGroup title="Select One">
        <Radio
          value="a"
          id="r1"
          name="value 1"
          checked={radio === 'a'}
          onChange={handleChange}
        />
        <Radio
          value="b"
          id="r2"
          name="value 2"
          checked={radio === 'b'}
          onChange={handleChange}
        />
        <Radio
          value="c"
          id="r3"
          name="value 3"
          checked={radio === 'c'}
          onChange={handleChange}
        />
      </RadioGroup>
    </div>
  )
}

export default RadioExample
```
