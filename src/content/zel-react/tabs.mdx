# Tabs

The Zeppelin Tabs requires the use of a theme.


<div style={{ marginBottom: 30 }}>
  <TabsExample />
</div>

```
import React, {useState} from 'react'
import Tabs from '@zlab-de/zel-react/Tabs'
import Tab from '@zlab-de/zel-react/Tab'
import ZepIconEssentialStar from '@zlab-de/zel-react-icons/ZepIconEssentialStar'

function MyComponent(){
    const [value, setValue] = useState(0)

    function handleChange(newValue) {
        setValue(newValue)
    }

    return(
        <div>
           <Tabs value={value} onClick={handleChange}>
                <Tab label="Tab 1" />
                <Tab label="Tab 2" />
            </Tabs>
            {value === 0 && <p>This is Tab 1</p>}
            {value === 1 && <p>This is Tab 2</p>}

            <Tabs value={value} onClick={handleChange} size="small">
                <Tab label="Tab 1" icon={<ZepIconEssentialStar />} />
                <Tab label="Tab 2" icon={<ZepIconEssentialStar />} />
            </Tabs>
            {value === 0 && <p>This is Tab 1</p>}
            {value === 1 && <p>This is Tab 2</p>}
        </div>
    )
}

export default MyComponent
```
