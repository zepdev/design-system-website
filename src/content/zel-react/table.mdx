# Table

Tables are a nice way to organize and present static data in a tabular format. They make it easier for the user to compare and scan for information.

```
import React from 'react'
import Table from '@zlab-de/zel-react/Table'
import TableHeader from '@zlab-de/zel-react/TableHeader'
import TableCell from '@zlab-de/zel-react/TableCell'
import TableBody from '@zlab-de/zel-react/TableBody'
import TableRow from '@zlab-de/zel-react/TableRow'

function MyComponent(){
    return(
        <div>
            <Table title="MyTableComponent">
            <TableHeader>
                {[1, 2].map(col => (
                <TableCell key={`header_${col}`}>Header {col}</TableCell>
                ))}
            </TableHeader>
            <TableBody>
                {[1, 2].map((row, idx) => (
                <TableRow key={`row_${row}`} color={idx % 2 === 0 ? 'white' : 'gray'}>
                    {[1, 2].map((cell, index) => (
                    <TableCell
                        key={`cell_${cell}`}
                        scope={index === 0 ? 'row' : null}
                        component={index === 0 ? 'th' : null}
                    >
                        Cell {cell}
                    </TableCell>
                    ))}
                </TableRow>
                ))}
            </TableBody>
            </Table>
        </div>
    )
}

export default MyComponent
```
