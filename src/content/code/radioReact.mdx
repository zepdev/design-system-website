```jsx
// Form
import React, { useState } from 'react'
import RadioGroup from '../radio/RadioGroup'
import Radio from '../radio/Radio'

function MyForm() {
   const [radio, setRadio] = React.useState('a')

  function handleChange(event) {
    setRadio(event.target.value)
  }
  return (
    <RadioGroup title="Select One">
        <Radio value="a" name="value 1" checked={radio === 'a'} onChange={handleChange} />
        <Radio value="b" name="value 2" checked={radio === 'b'} onChange={handleChange} />
        <Radio value="c" name="value 3" checked={radio === 'c'} onChange={handleChange} />
    </RadioGroup>
  )
}

export default MyForm

// Radio Group
import React from 'react'
import PropTypes from 'prop-types'
import classnames from 'classnames'

const RadioGroup = ({ title, children, className: classNameProp }) => (
  <fieldset
    role="group"
    aria-labelledby={`radio_${ title }`}
    className={classnames('zep-typo--normal-3', classNameProp)}
  >
    <legend id={`radio_${ title }`}>{title}</legend>
    {children}
  </fieldset>
)

RadioGroup.propTypes = {
  title: PropTypes.string,
  className: PropTypes.object,
  children: PropTypes.array,
}

export default RadioGroup

// Radio
import React from 'react'
import PropTypes from 'prop-types'
import classnames from 'classnames'

const Radio = ({ checked, onChange, value, name, className: classNameProp }) => (
  <label className={classnames('zep-radio', classNameProp)}>
    <input
      type="radio"
      name={name}
      className="zep-radio__input"
      value={value}
      checked={checked}
      onChange={onChange}
    />
    <span className="zep-radio__indicator" />
    <span className="zep-radio__label">{name}</span>
  </label>
)

Radio.propTypes = {
  className: PropTypes.object,
  checked: PropTypes.bool.isRequired,
  onChange: PropTypes.func.isRequired,
  value: PropTypes.string.isRequired,
  name: PropTypes.string.isRequired
}

export default Radio
```
